type Query {
  sample_tables(filter: sample_table_filter): [sample_table]
}

"""SAMPLE_TABLE"""
type sample_table {
  string_field: String!
  int_field: Int!
  float_field: Float!
  bool_field: Boolean!
  date_field: Date!
  datetime_field: DateTime!
  time_field: Time!
  bytes_field: Bytes!
  json_field: JSON!
  enum_field: SampleEnum!
  nullable_field: String
}

scalar Date

scalar DateTime

scalar Time

scalar Bytes

scalar JSON

enum SampleEnum {
  ODD
  EVEN
}

input sample_table_filter {
  string_field: StringFilter
  int_field: IntFilter
  float_field: FloatFilter
  bool_field: BoolFilter
  date_field: DateFilter
  datetime_field: DateTimeFilter
  time_field: TimeFilter
  enum_field: SampleEnum_filter
}

input StringFilter {
  eq: String
  ne: String
  contains: String
  startswith: String
  endswith: String
  in: [String]
}

input IntFilter {
  eq: Int
  ne: Int
  gt: Int
  ge: Int
  lt: Int
  le: Int
  in: [Int]
}

input FloatFilter {
  eq: Float
  ne: Float
  gt: Float
  ge: Float
  lt: Float
  le: Float
  in: [Float]
}

input BoolFilter {
  eq: Boolean
  ne: Boolean
}

input DateFilter {
  eq: Date
  ne: Date
  gt: Date
  ge: Date
  lt: Date
  le: Date
  in: [Date]
}

input DateTimeFilter {
  eq: DateTime
  ne: DateTime
  gt: DateTime
  ge: DateTime
  lt: DateTime
  le: DateTime
  in: [DateTime]
}

input TimeFilter {
  eq: Time
  ne: Time
  gt: Time
  ge: Time
  lt: Time
  le: Time
  in: [Time]
}

input SampleEnum_filter {
  eq: SampleEnum
  ne: SampleEnum
  in: [SampleEnum]
}
