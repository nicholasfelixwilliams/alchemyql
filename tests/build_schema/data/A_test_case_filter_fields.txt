type Query {
  sample_tables(filter: sample_table_filter): [sample_table]
}

type sample_table {
  string_field: String!
  int_field: Int!
  float_field: Float!
  bool_field: Boolean!
  date_field: Date!
  datetime_field: DateTime!
  time_field: Time!
  bytes_field: Bytes!
  json_field: JSON!
  enum_field: SampleEnum!
  nullable_field: String
}

scalar Date

scalar DateTime

scalar Time

scalar Bytes

scalar JSON

enum SampleEnum {
  ODD
  EVEN
}

input sample_table_filter {
  string_field: StringFilter
}

input StringFilter {
  eq: String
  ne: String
  contains: String
  startswith: String
  endswith: String
  in: [String]
}
